00082 QPainterPath KnotGraph::build()
00083 {
00084     knot_curve_style* pcs = knot_curve_styler::style(curve_style);
00085 
00086     QPainterPath path;
00087 
00088     foreach(Edge*e,edges)
00089     {
00090         e->reset();
00091         QList<QLineF> connected = e->connected(handle_length,crossing_distance);
00092         foreach ( QLineF l, connected )
00093         {
00094             path.moveTo(l.p1());
00095             path.lineTo(l.p2());
00096         }
00097     }
00098 
00099     while ( !edges.empty() )
00100     {
00101         Edge* e = edges.front();
00102         Edge::handle_type handle = e->not_traversed();
00103         if ( handle == Edge::NOHANDLE )
00104         {
00105             traversed_edges.push_back(e);
00106             edges.pop_front();
00107         }
00108         else
00109         {
00110             if ( ! e->traversed(handle) )
00111             {
00112                 e->traverse(handle);
00113 
00114                 Node* n = e->vertex1();
00115                 if ( handle == Edge::TOPRIGHT || handle == Edge::BOTTOMRIGHT )
00116                     n = e->vertex2();
00117 
00118 
00119                 TraversalInfo ti = n->next_edge(e,handle);
00120 
00121                 pcs->draw_joint(path,n,ti,cusp_angle,handle_length,crossing_distance);
00122 
00123                 ti.edge_out->traverse(ti.handle_out);
00124             }
00125         }
00126     }
00127 
00128 
00129     edges = traversed_edges;
00130     traversed_edges.clear();
00131 
00132     setPath(stroker.createStroke(path).simplified());
00133 
00134     return path;
00135 }
