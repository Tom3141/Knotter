00033 void snapping_grid::snap(QPointF &p)
00034 {
00035     /// \todo find a way to draw the grid (lines or dots?)
00036     /// \todo find a cool way to configure the origin ( cool way != typing coordinates )
00037 
00038     if ( !enabled )
00039         return;
00040 
00041     if ( shape == SQUARE )
00042     {
00043         /**
00044             Square grid, simple enough
00045         */
00046         p /= size;
00047         p.setX(qRound64(p.x()));
00048         p.setY(qRound64(p.y()));
00049         p *= size;
00050         p += origin;
00051     }
00052     else if ( shape == TRIANGLE )
00053     {
00054         /**
00055             Triangular grid, find intersection of line1 and line2
00056             line2 is horizontal: y = n2 * y_factor + origin.y
00057             y_factor is the height of the triangle sqrt(3)/2 * size.
00058             line1 has a 30deg slope, and x offset of n1*size + origin.x
00059             y = sqrt(3) * ( x + n1*size + origin.x )
00060             They intersect in the point where y is constrained by line2,
00061             x = size/2 * n2 + size * n1
00062 
00063             if you think of p as (x,y)+offset, you can get the expressions used
00064             below to get (x,y) from (p.x,p.y).
00065             The rounding p/size removes the offset as |offset.x| < size
00066         */
00067         double y_factor = size * sqrt(3.0)/2.0;
00068         qint64 n2 = qRound64(p.y()/y_factor);
00069         p.setY(n2*y_factor+origin.y());
00070         qint64 n1 = qRound64 ( p.x()/size - n2/2.0 );
00071         p.setX(size*(n2/2.0+n1)+origin.y());
00072     }
00073 }
