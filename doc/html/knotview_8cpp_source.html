01007 void KnotView::merge_selected()
01008 {
01009 
01010     if ( mode == EDIT_NODE )
01011     {
01012 
01013         QPointF pos; // average position
01014 
01015         node_list links; // nodes linked to selection
01016 
01017         QList<QGraphicsItem*> selection = scene()->selectedItems();
01018 
01019         if ( selection.count() < 2 )
01020             return;
01021 
01022         undo_stack.beginMacro(tr("Merge Nodes"));
01023 
01024         foreach( QGraphicsItem* node, selection )
01025         {
01026             Node* n = dynamic_cast<Node*>(node);
01027             if ( n )
01028             {
01029                 pos += node->pos();
01030                 node_list adjacency_list;
01031                 foreach ( Edge* e, n->links() )
01032                 {
01033                     Node *o = e->other(n);
01034                     adjacency_list.push_back(o);
01035                     if ( !scene()->selectedItems().contains(o) )
01036                         links.push_back(o);
01037                 }
01038                 remove_node(n,adjacency_list);
01039             }
01040         }
01041 
01042         add_node(pos/selection.count(),links);
01043         last_node->setSelected(true);
01044 
01045         undo_stack.endMacro();
01046 
01047     }
