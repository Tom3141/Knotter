01013 {
01014 
01015     if ( mode == EDIT_NODE )

01017

01019

01022         QList<QGraphicsItem*> selection = scene()->selectedItems();
01023 
01024         if ( selection.count() < 2 )
01025             return;
01026 
01027         undo_stack.beginMacro(tr("Merge Nodes"));
01028 
01029         foreach( QGraphicsItem* node, selection )
01030         {
01031             Node* n = dynamic_cast<Node*>(node);
01032             if ( n )
01033             {
01034                 pos += node->pos();
01035                 node_list adjacency_list;
01036                 foreach ( Edge* e, n->links() )
01037                 {
01038                     Node *o = e->other(n);
01039                     adjacency_list.push_back(o);
01040                     if ( !scene()->selectedItems().contains(o) )
01041                         links.push_back(o);
01042                 }
01043                 remove_node(n,adjacency_list);
01044             }
01045         }
01046 
01047         add_node(pos/selection.count(),links);
01048         last_node->setSelected(true);

01050         undo_stack.endMacro();
